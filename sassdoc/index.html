	<div class="handler-item">
		<a class="dropdown-item" href="link">
			<i class="ficon fa-user"></i>
			<span class="text">Open dropdown</span>
		</a>
		<div class="handler">
			<i class="ficon fa-caret-down"></i> <!-- /*font-awesome icon*/ -->
		</div>
	</div>
</div>
